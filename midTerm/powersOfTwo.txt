
public class Main {
    public static void main(String[] args) {

        System.out.println(isPowerOfTwo(5));
        System.out.println(isPowerOfTwo(8));

    }

    public static boolean isPowerOfTwo(int n) {
        if (n <= 0) {
            return false;
        }
        int x = (int)Math.pow(2, 0);
        int i = 0;
        while (x < n) {
            x = (int)Math.pow(2, i);
            i++;
        }
        return x == n;
    }

}
